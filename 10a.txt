#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/stat.h>
#include <fcntl.h>

int main() {
    pid_t pid;

    
    pid = fork();
    if (pid < 0) {
        exit(1);
    }
    if (pid > 0) {
        
        exit(0);
    }

    
    if (setsid() < 0) {
        exit(1);
    }

    
    umask(0);

   
    chdir("/");

    
    close(STDIN_FILENO);
    close(STDOUT_FILENO);
    close(STDERR_FILENO);

    
    while (1) {
        
        sleep(5);
    }

    return 0;
}
