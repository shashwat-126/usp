#include <stdio.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>

int main() {
    pid_t pid = fork();

    if (pid < 0) {
        perror("fork failed");
        return 1;
    }

    if (pid == 0) {
        
        printf("Child Process (PID: %d)\n", getpid());

        
        execl("./add", "add", "10", "20", NULL);

        perror("execl failed");
    } else {
        
        printf("Parent Process (PID: %d)\n", getpid());

        wait(NULL);  
        printf("Child process completed\n");
    }

    return 0;
}
