#include <stdio.h>

extern char **environ;

int main(int argc, char *argv[]) {

    printf("\n--- Arguments ---\n");
    for (int i = 0; i < argc; i++) {
        printf("argv[%d] = %s\n", i, argv[i]);
    }

    printf("\n--- Environment Variables ---\n");
    for (int i = 0; environ[i] != NULL; i++) {
        printf("%s\n", environ[i]);
    }

    return 0;
}
