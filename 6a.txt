#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
#include <string.h>

int main() {
    int fd, n;
    char buffer[100];

    printf("Enter number of characters to read: ");
    scanf("%d", &n);

    fd = open("sample.txt", O_RDWR);
    if (fd < 0) {
        perror("open failed");
        return 1;
    }

    
    int bytes = read(fd, buffer, n);
    if (bytes < 0) {
        perror("read failed");
        return 1;
    }

    
    lseek(fd, 0, SEEK_END);

    
    dup2(fd, STDOUT_FILENO);

   
    write(STDOUT_FILENO, buffer, bytes);

    printf("\nData appended successfully\n");

    close(fd);
    return 0;
}
