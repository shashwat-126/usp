int mysystem(const char *command) {

    pid = fork();

    if (pid < 0) {

✅ Concept
Use fcntl() with F_GETLK to check lock

💻 Code
#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/types.h>

int main() {
    int fd;
    struct flock lock;
    char buffer[51];

    fd = open("sample.txt", O_RDWR);
    if (fd < 0) {
        perror("open failed");
        return 1;
    }

    // Move to end to find size
    off_t size = lseek(fd, 0, SEEK_END);

    // Define last 100 bytes region
    lock.l_type = F_WRLCK;
    lock.l_whence = SEEK_SET;
    lock.l_start = size - 100;
    lock.l_len = 100;

    // Check if locked
    if (fcntl(fd, F_GETLK, &lock) == -1) {
        perror("fcntl error");
        return 1;
    }

    if (lock.l_type != F_UNLCK) {
        printf("Region is locked by process with PID: %d\n", lock.l_pid);
    } else {
        printf("Region is not locked. Locking now...\n");

        // Apply exclusive lock
        lock.l_type = F_WRLCK;
        fcntl(fd, F_SETLK, &lock);

        // Read last 50 bytes
        lseek(fd, size - 50, SEEK_SET);
        read(fd, buffer, 50);
        buffer[50] = '\0';

        printf("Last 50 bytes:\n%s\n", buffer);

        // Unlock region
        lock.l_type = F_UNLCK;
        fcntl(fd, F_SETLK, &lock);

        printf("Region unlocked\n");
    }

    close(fd);
    return 0;
}
▶️ Important Notes

F_SETLK → sets/unsets lock

F_WRLCK → exclusive (write) lock

l_pid → gives process holding lock

    pid_t pid;

    // Step 1: Fork
    pid = fork();
    if (pid < 0) {

    close(STDERR_FILENO);

💻 Code
#include <stdio.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>

int main() {
    pid_t pid1, pid2;
    int status;

    // First child
    pid1 = fork();
    if (pid1 == 0) {
        printf("Child 1: PID = %d\n", getpid());
        sleep(2);
        printf("Child 1 exiting\n");
        return 1;
    }

    // Second child
    pid2 = fork();
    if (pid2 == 0) {
        printf("Child 2: PID = %d\n", getpid());
        sleep(4);
        printf("Child 2 exiting\n");
        return 2;
    }

    // Parent process
    printf("Parent PID: %d\n", getpid());

    // wait() → waits for any child
    wait(&status);
    printf("Parent: One child finished (wait)\n");

    // waitpid() → wait for specific child (pid2)
    waitpid(pid2, &status, 0);
    printf("Parent: Child 2 finished (waitpid)\n");

    return 0;
}
