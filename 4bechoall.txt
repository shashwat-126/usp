#include <stdio.h>

int main(int argc, char *argv[]) {

    printf("Total arguments received: %d\n", argc);

    for (int i = 0; i < argc; i++) {
        printf("Argument %d: %s\n", i, argv[i]);
    }

    return 0;
}
