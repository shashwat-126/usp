#include <stdio.h>
#include <unistd.h>
#include <sys/types.h>

int main() {
    pid_t pid = fork();

    if (pid < 0) {
        perror("Fork failed");
        return 1;
    }

    if (pid == 0) {
        // Child executes interpreter file
        execl("./interpreter.sh", "interpreter.sh", "HelloFromChild", NULL);
        perror("execl failed");
    }

    return 0;
}
