#include <stdio.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>

int main() {
    pid_t pid = fork();

    if (pid < 0) {
        perror("Fork failed");
        return 1;
    }

    if (pid == 0) {
        // Child process
        printf("Child Process\n");
        printf("Child PID: %d, Parent PID: %d\n", getpid(), getppid());

        // Execute access program
        execl("./access_check", "access_check", "sample.txt", NULL);

        perror("execl failed");
    } 
    else {
        // Parent process
        printf("Parent Process\n");
        printf("Parent PID: %d, Child PID: %d\n", getpid(), pid);

        wait(NULL);  // wait for child
        printf("Child process finished\n");
    }

    return 0;
}
